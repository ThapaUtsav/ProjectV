CREATE TABLE admin (
    account_number VARCHAR(50) PRIMARY KEY,
    Grp_n varchar(225),  
    Email VARCHAR(225),                 
    PH_num VARCHAR(10),                 
    PassW VARCHAR(225)                  
);


CREATE TABLE admins_info (
    id INT AUTO_INCREMENT PRIMARY KEY,  
    admin_name VARCHAR(225),                 
    username VARCHAR(255),                 
    dob DATE,
    street VARCHAR(255),
    city VARCHAR(100),
    state VARCHAR(100),
    account_number VARCHAR(50),  -- Add this column
    FOREIGN KEY (account_number) REFERENCES admin(account_number) 
);
CREATE TABLE users (  
    account_num VARCHAR(50) PRIMARY KEY,  
    name VARCHAR(255) NOT NULL,
    username VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    dob DATE,
    phone VARCHAR(15),
    street VARCHAR(255),
    city VARCHAR(100),
    state VARCHAR(100),
    acc_date DATE, 
    FOREIGN KEY (account_num) REFERENCES admin(account_number)  
);
