CREATE TABLE admin (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Grp_n VARCHAR(50),
    Email VARCHAR(50),                 
    PH_num VARCHAR(10),                 
    PassW VARCHAR(225)                  
);

CREATE TABLE user_profiles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    email VARCHAR(100) NOT NULL,
    dob DATE,
    phone VARCHAR(20),
    street VARCHAR(255),
    city VARCHAR(50),
    state VARCHAR(50),
    zip VARCHAR(10),
    accountType ENUM('saving', 'loan') NOT NULL,
    accDate DATE
);
